app.js

var App = Ember.Application.create({
  LOG_TRANSITIONS: true
});
App.Router.map(function() {
  this.route('credits', { path: '/thanks' });
  this.route('about');
});
App.IndexController = Ember.Controller.extend({
  productsCount: 6,
  logo: 'images/logo-small.png',
  time: function() {
    return (new Date()).toDateString();
  }.property()
});
App.AboutController = Ember.Controller.extend({
  contactName: 'Anostagia',
  avatar: 'images/avatar.png',
  open: function() {
    return ((new Date()).getDay() === 0) ? "Closed" : "Open";
  }.property()
});

4. In order to pass these products to our view, we’ll need to create another object. Create the object that will pass the array of products as the model.

Answer: 

App.ProductsRoute = Ember.Route.extend({
  model: function() {
    return App.PRODUCTS;
  }
});

5. We’re all set to show the products! Create a template for our products page, and in it loop over each of them. Output their title and image. Don’t worry too much about styling, we’ll clean that up in the next challenge.

Answer: 

<script type='text/x-handlebars' data-template-name='products'>
  <h2>Products</h2>
  {{#each}}
    <h2>{{title}}</h2>
    <img {{bind-attr src='image'}} />
  {{/each}}
</script>
